    if guess in display:
        level -= 1

        Blocker = True 
        thingy()
        
    if Blocker == False: 
        for position in range(word_length):
            letter = chosen_word[position]
            if letter == guess:
                display[position] = letter
                clear()

    if guess not in chosen_word:
        clear()

        level -= 1 
        thingy()
    

    if "_" not in display:
        clear()
        print("You've won!")
        time.sleep(1)

        break
